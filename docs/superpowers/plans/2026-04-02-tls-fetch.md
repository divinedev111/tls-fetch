```go
// Package header provides ordered HTTP header types that preserve insertion order.
package header

import (
	"net/http"
	"strings"
)

// Ordered is an ordered list of header key-value pairs.
// Unlike http.Header (a map), it preserves the order headers were added.
type Ordered []HeaderPair

// HeaderPair is a single header name-value pair.
// Index 0 is the header name, index 1 is the value.
type HeaderPair [2]string

// Get returns the value for the given header name (case-insensitive).
// Returns empty string if not found.
func (o Ordered) Get(name string) string {
	for _, pair := range o {
		if strings.EqualFold(pair[0], name) {
			return pair[1]
		}
	}
	return ""
}

// Set replaces the value for the given header name, or appends if not found.
// Returns the modified Ordered (may be a new slice if appended).
func (o Ordered) Set(name, value string) Ordered {
	for i, pair := range o {
		if strings.EqualFold(pair[0], name) {
			o[i] = HeaderPair{pair[0], value}
			return o
		}
	}
	return append(o, HeaderPair{name, value})
}

// Del removes the header with the given name (case-insensitive).
// Returns the modified Ordered.
func (o Ordered) Del(name string) Ordered {
	for i, pair := range o {
		if strings.EqualFold(pair[0], name) {
			return append(o[:i], o[i+1:]...)
		}
	}
	return o
}

// Clone returns a deep copy that does not alias the original.
func (o Ordered) Clone() Ordered {
	if o == nil {
		return nil
	}
	clone := make(Ordered, len(o))
	copy(clone, o)
	return clone
}

// ToHTTPHeader converts to a standard http.Header (loses ordering).
func (o Ordered) ToHTTPHeader() http.Header {
	h := make(http.Header, len(o))
	for _, pair := range o {
		h.Set(pair[0], pair[1])
	}
	return h
}

// Keys returns the header names in order.
func (o Ordered) Keys() []string {
	keys := make([]string, len(o))
	for i, pair := range o {
		keys[i] = pair[0]
	}
	return keys
}
```

```go
package header

import (
	"bytes"
	"strings"
)

var headerEnd = []byte("\r\n\r\n")
var crlf = []byte("\r\n")

// ReorderHTTP1Headers takes raw HTTP/1.1 request bytes (request line + headers + body)
// and reorders the headers according to the given order.
// Headers not in the order list are appended after ordered headers.
// The request line and body are not modified.
func ReorderHTTP1Headers(raw []byte, order []string) []byte {
	if len(order) == 0 {
		return raw
	}

	// Split at \r\n\r\n to separate headers from body
	idx := bytes.Index(raw, headerEnd)
	if idx == -1 {
		return raw
	}

	headerSection := raw[:idx]
	body := raw[idx:] // includes \r\n\r\n and any body bytes

	// Split header section into lines
	lines := bytes.Split(headerSection, crlf)
	if len(lines) < 2 {
		return raw
	}

	requestLine := lines[0]
	headerLines := lines[1:]

	// Parse headers into name→full line map
	type headerLine struct {
		name string
		line []byte
	}
	parsed := make([]headerLine, 0, len(headerLines))
	for _, line := range headerLines {
		if len(line) == 0 {
			continue
		}
		colonIdx := bytes.IndexByte(line, ':')
		if colonIdx == -1 {
			continue
		}
		name := string(bytes.TrimSpace(line[:colonIdx]))
		parsed = append(parsed, headerLine{name: name, line: line})
	}

	// Build ordered output
	var buf bytes.Buffer
	buf.Grow(len(raw))
	buf.Write(requestLine)
	buf.Write(crlf)

	used := make(map[int]bool, len(parsed))

	// First: write headers in the specified order
	for _, orderedName := range order {
		for i, h := range parsed {
			if !used[i] && strings.EqualFold(h.name, orderedName) {
				buf.Write(h.line)
				buf.Write(crlf)
				used[i] = true
				break
			}
		}
	}

	// Then: append remaining headers in their original order
	for i, h := range parsed {
		if !used[i] {
			buf.Write(h.line)
			buf.Write(crlf)
		}
	}

	buf.Write(body)
	return buf.Bytes()
}
```

```go
// Package internal contains unexported implementation details.
package internal

import (
	"net"
	"sync"
	"time"
)

// PoolConfig configures connection pool behavior.
type PoolConfig struct {
	MaxConnsPerHost int
	MaxIdleConns    int
	IdleTimeout     time.Duration
	TTL             time.Duration
}

// PoolStats contains pool statistics.
type PoolStats struct {
	TotalIdle int
	Hosts     int
}

type idleConn struct {
	conn      net.Conn
	idleSince time.Time
}

// Pool manages idle connections grouped by host.
type Pool struct {
	mu     sync.Mutex
	conns  map[string][]idleConn
	config PoolConfig
	closed bool
}

// NewPool creates a connection pool with the given config.
func NewPool(cfg PoolConfig) *Pool {
	return &Pool{
		conns:  make(map[string][]idleConn),
		config: cfg,
	}
}

// Get retrieves an idle connection for the host, or nil if none available.
// Expired connections are closed and skipped.
func (p *Pool) Get(host string) net.Conn {
	p.mu.Lock()
	defer p.mu.Unlock()

	if p.closed {
		return nil
	}

	conns := p.conns[host]
	now := time.Now()

	for len(conns) > 0 {
		// Take from the end (LIFO — most recently used)
		last := conns[len(conns)-1]
		conns = conns[:len(conns)-1]

		if p.config.IdleTimeout > 0 && now.Sub(last.idleSince) > p.config.IdleTimeout {
			last.conn.Close()
			continue
		}

		p.conns[host] = conns
		return last.conn
	}

	delete(p.conns, host)
	return nil
}

// Put returns a connection to the pool. If the pool for this host is full,
// the connection is closed immediately.
func (p *Pool) Put(host string, conn net.Conn) {
	p.mu.Lock()
	defer p.mu.Unlock()

	if p.closed {
		conn.Close()
		return
	}

	conns := p.conns[host]

	if len(conns) >= p.config.MaxConnsPerHost {
		conn.Close()
		return
	}

	p.conns[host] = append(conns, idleConn{
		conn:      conn,
		idleSince: time.Now(),
	})
}

// Stats returns current pool statistics.
func (p *Pool) Stats() PoolStats {
	p.mu.Lock()
	defer p.mu.Unlock()

	total := 0
	for _, conns := range p.conns {
		total += len(conns)
	}

	return PoolStats{
		TotalIdle: total,
		Hosts:     len(p.conns),
	}
}

// Close closes all pooled connections and prevents future use.
func (p *Pool) Close() {
	p.mu.Lock()
	defer p.mu.Unlock()

	p.closed = true
	for host, conns := range p.conns {
		for _, ic := range conns {
			ic.conn.Close()
		}
		delete(p.conns, host)
	}
}
```

```go
package tlsfetch

import (
	"encoding/json"
	"fmt"
	"os"

	utls "github.com/refraction-networking/utls"
)

// jsonProfile is the on-disk representation of a BrowserProfile.
type jsonProfile struct {
	Name    string          `json:"name"`
	TLS     jsonTLSConfig   `json:"tls"`
	H2      jsonH2Config    `json:"h2"`
	Headers jsonHeadersConfig `json:"headers"`
}

type jsonTLSConfig struct {
	ClientHelloID string `json:"client_hello_id"`
}

type jsonH2Config struct {
	Settings         []jsonH2Setting `json:"settings"`
	WindowUpdate     uint32          `json:"window_update"`
	PseudoHeaderOrder []string       `json:"pseudo_header_order"`
}

type jsonH2Setting struct {
	ID    uint16 `json:"id"`
	Value uint32 `json:"value"`
}

type jsonHeadersConfig struct {
	Order []string `json:"order"`
}

// clientHelloIDMap maps JSON string names to uTLS ClientHelloID values.
var clientHelloIDMap = map[string]utls.ClientHelloID{
	"HelloChrome_131":  utls.HelloChrome_131,
	"HelloChrome_Auto": utls.HelloChrome_Auto,
	"HelloFirefox_120": utls.HelloFirefox_120,
	"HelloFirefox_Auto": utls.HelloFirefox_Auto,
	"HelloSafari_Auto": utls.HelloSafari_Auto,
	"HelloEdge_Auto":   utls.HelloEdge_Auto,
	"HelloEdge_85":     utls.HelloEdge_85,
	"HelloCustom":      utls.HelloCustom,
}

// LoadProfileFromFile reads a JSON file and returns a BrowserProfile.
func LoadProfileFromFile(path string) (BrowserProfile, error) {
	data, err := os.ReadFile(path)
	if err != nil {
		return BrowserProfile{}, fmt.Errorf("tlsfetch: read profile: %w", err)
	}
	return LoadProfileFromJSON(data)
}

// LoadProfileFromJSON parses JSON bytes into a BrowserProfile.
func LoadProfileFromJSON(data []byte) (BrowserProfile, error) {
	var jp jsonProfile
	if err := json.Unmarshal(data, &jp); err != nil {
		return BrowserProfile{}, fmt.Errorf("tlsfetch: parse profile JSON: %w", err)
	}

	if jp.Name == "" {
		return BrowserProfile{}, &ErrInvalidProfile{Name: "", Reason: "name is required"}
	}

	helloID, ok := clientHelloIDMap[jp.TLS.ClientHelloID]
	if !ok && jp.TLS.ClientHelloID != "" {
		return BrowserProfile{}, &ErrInvalidProfile{
			Name:   jp.Name,
			Reason: fmt.Sprintf("unknown client_hello_id: %q", jp.TLS.ClientHelloID),
		}
	}

	settings := make([]H2Setting, len(jp.H2.Settings))
	for i, s := range jp.H2.Settings {
		settings[i] = H2Setting{ID: s.ID, Value: s.Value}
	}

	var pseudoHeaders [4]string
	for i, ph := range jp.H2.PseudoHeaderOrder {
		if i >= 4 {
			break
		}
		pseudoHeaders[i] = ph
	}

	return BrowserProfile{
		Name:              jp.Name,
		ClientHelloID:     helloID,
		H2Settings:        settings,
		H2WindowUpdate:    jp.H2.WindowUpdate,
		PseudoHeaderOrder: pseudoHeaders,
		HeaderOrder:       jp.Headers.Order,
	}, nil
}
```

```go
// Package proxy provides network dialers for HTTP and SOCKS5 proxies.
package proxy

import (
	"context"
	"fmt"
	"net"
	"net/url"
	"time"
)

// Dialer establishes a TCP connection, optionally through a proxy.
type Dialer interface {
	DialContext(ctx context.Context, network, addr string) (net.Conn, error)
}

// FromURL returns a Dialer for the given proxy URL.
// Nil URL returns a direct dialer (no proxy).
func FromURL(u *url.URL) (Dialer, error) {
	if u == nil {
		return &directDialer{}, nil
	}

	switch u.Scheme {
	case "http", "https":
		return &httpDialer{addr: u.Host, user: u.User}, nil
	case "socks5", "socks5h":
		return &socks5Dialer{addr: u.Host, user: u.User}, nil
	default:
		return nil, fmt.Errorf("proxy: unsupported scheme %q", u.Scheme)
	}
}

// directDialer connects directly without a proxy.
type directDialer struct{}

func (d *directDialer) DialContext(ctx context.Context, network, addr string) (net.Conn, error) {
	nd := &net.Dialer{Timeout: 30 * time.Second}
	return nd.DialContext(ctx, network, addr)
}
```

```go
package proxy

import (
	"bufio"
	"context"
	"encoding/base64"
	"fmt"
	"net"
	"net/http"
	"net/url"
	"time"
)

// httpDialer connects through an HTTP CONNECT proxy.
type httpDialer struct {
	addr string
	user *url.Userinfo
}

func (d *httpDialer) DialContext(ctx context.Context, network, addr string) (net.Conn, error) {
	nd := &net.Dialer{Timeout: 30 * time.Second}
	conn, err := nd.DialContext(ctx, "tcp", d.addr)
	if err != nil {
		return nil, fmt.Errorf("proxy: dial proxy %s: %w", d.addr, err)
	}

	req := &http.Request{
		Method: http.MethodConnect,
		URL:    &url.URL{Opaque: addr},
		Host:   addr,
		Header: make(http.Header),
	}

	if d.user != nil {
		username := d.user.Username()
		password, _ := d.user.Password()
		cred := base64.StdEncoding.EncodeToString([]byte(username + ":" + password))
		req.Header.Set("Proxy-Authorization", "Basic "+cred)
	}

	if err := req.Write(conn); err != nil {
		conn.Close()
		return nil, fmt.Errorf("proxy: CONNECT write: %w", err)
	}

	resp, err := http.ReadResponse(bufio.NewReader(conn), req)
	if err != nil {
		conn.Close()
		return nil, fmt.Errorf("proxy: CONNECT read: %w", err)
	}
	resp.Body.Close()

	if resp.StatusCode != http.StatusOK {
		conn.Close()
		return nil, fmt.Errorf("proxy: CONNECT status %d", resp.StatusCode)
	}

	return conn, nil
}
```

```go
package proxy

import (
	"context"
	"encoding/binary"
	"fmt"
	"io"
	"net"
	"net/url"
	"strconv"
	"time"
)

// socks5Dialer connects through a SOCKS5 proxy.
type socks5Dialer struct {
	addr string
	user *url.Userinfo
}

func (d *socks5Dialer) DialContext(ctx context.Context, network, addr string) (net.Conn, error) {
	nd := &net.Dialer{Timeout: 30 * time.Second}
	conn, err := nd.DialContext(ctx, "tcp", d.addr)
	if err != nil {
		return nil, fmt.Errorf("proxy: dial SOCKS5 %s: %w", d.addr, err)
	}

	if err := d.handshake(conn, addr); err != nil {
		conn.Close()
		return nil, err
	}

	return conn, nil
}

func (d *socks5Dialer) handshake(conn net.Conn, targetAddr string) error {
	// Greeting: version 5, auth methods
	hasAuth := d.user != nil
	if hasAuth {
		conn.Write([]byte{0x05, 0x02, 0x00, 0x02}) // no-auth + username/password
	} else {
		conn.Write([]byte{0x05, 0x01, 0x00}) // no-auth only
	}

	// Server response
	buf := make([]byte, 2)
	if _, err := io.ReadFull(conn, buf); err != nil {
		return fmt.Errorf("proxy: SOCKS5 greeting: %w", err)
	}
	if buf[0] != 0x05 {
		return fmt.Errorf("proxy: SOCKS5 version %d, want 5", buf[0])
	}

	// Handle auth if server requests it
	if buf[1] == 0x02 {
		if !hasAuth {
			return fmt.Errorf("proxy: SOCKS5 requires auth but no credentials")
		}
		username := d.user.Username()
		password, _ := d.user.Password()

		authReq := []byte{0x01, byte(len(username))}
		authReq = append(authReq, []byte(username)...)
		authReq = append(authReq, byte(len(password)))
		authReq = append(authReq, []byte(password)...)
		conn.Write(authReq)

		authResp := make([]byte, 2)
		if _, err := io.ReadFull(conn, authResp); err != nil {
			return fmt.Errorf("proxy: SOCKS5 auth: %w", err)
		}
		if authResp[1] != 0x00 {
			return fmt.Errorf("proxy: SOCKS5 auth failed")
		}
	} else if buf[1] != 0x00 {
		return fmt.Errorf("proxy: SOCKS5 unsupported auth method %d", buf[1])
	}

	// CONNECT request
	host, portStr, err := net.SplitHostPort(targetAddr)
	if err != nil {
		return fmt.Errorf("proxy: bad target addr %q: %w", targetAddr, err)
	}
	port, err := strconv.Atoi(portStr)
	if err != nil {
		return fmt.Errorf("proxy: bad port %q: %w", portStr, err)
	}

	connectReq := []byte{0x05, 0x01, 0x00} // version, connect, reserved

	// Address type
	if ip := net.ParseIP(host); ip != nil {
		if ip4 := ip.To4(); ip4 != nil {
			connectReq = append(connectReq, 0x01) // IPv4
			connectReq = append(connectReq, ip4...)
		} else {
			connectReq = append(connectReq, 0x04) // IPv6
			connectReq = append(connectReq, ip...)
		}
	} else {
		connectReq = append(connectReq, 0x03) // Domain
		connectReq = append(connectReq, byte(len(host)))
		connectReq = append(connectReq, []byte(host)...)
	}

	portBytes := make([]byte, 2)
	binary.BigEndian.PutUint16(portBytes, uint16(port))
	connectReq = append(connectReq, portBytes...)

	conn.Write(connectReq)

	// Read CONNECT response (minimum 10 bytes for IPv4)
	resp := make([]byte, 10)
	if _, err := io.ReadFull(conn, resp); err != nil {
		return fmt.Errorf("proxy: SOCKS5 connect: %w", err)
	}
	if resp[1] != 0x00 {
		return fmt.Errorf("proxy: SOCKS5 connect error code %d", resp[1])
	}

	return nil
}
```

```go
package tlsfetch

import (
	"fmt"
	"strconv"
	"strings"

	utls "github.com/refraction-networking/utls"
)

// ProfileFromJA3 parses a JA3 fingerprint string and returns a BrowserProfile
// with a custom ClientHelloSpec.
//
// JA3 format: SSLVersion,Ciphers,Extensions,EllipticCurves,EllipticCurvePointFormats
// Example: 771,4865-4866-4867,0-23-65281-10-11,29-23-24,0
func ProfileFromJA3(ja3 string) (BrowserProfile, error) {
	parts := strings.Split(ja3, ",")
	if len(parts) != 5 {
		return BrowserProfile{}, &ErrInvalidJA3{Input: ja3, Reason: fmt.Sprintf("expected 5 comma-separated fields, got %d", len(parts))}
	}

	// Parse TLS version
	version, err := strconv.ParseUint(parts[0], 10, 16)
	if err != nil {
		return BrowserProfile{}, &ErrInvalidJA3{Input: ja3, Reason: fmt.Sprintf("invalid TLS version: %s", parts[0])}
	}

	// Parse cipher suites
	ciphers, err := parseDashSeparatedUint16(parts[1])
	if err != nil {
		return BrowserProfile{}, &ErrInvalidJA3{Input: ja3, Reason: fmt.Sprintf("invalid ciphers: %v", err)}
	}
	ciphers = filterGREASE(ciphers)

	// Parse extension IDs
	extIDs, err := parseDashSeparatedUint16(parts[2])
	if err != nil {
		return BrowserProfile{}, &ErrInvalidJA3{Input: ja3, Reason: fmt.Sprintf("invalid extensions: %v", err)}
	}
	extIDs = filterGREASE(extIDs)

	// Parse elliptic curves
	curves, err := parseDashSeparatedUint16(parts[3])
	if err != nil {
		return BrowserProfile{}, &ErrInvalidJA3{Input: ja3, Reason: fmt.Sprintf("invalid curves: %v", err)}
	}
	curves = filterGREASE(curves)

	// Parse point formats
	pointFormats, err := parseDashSeparatedUint8(parts[4])
	if err != nil {
		return BrowserProfile{}, &ErrInvalidJA3{Input: ja3, Reason: fmt.Sprintf("invalid point formats: %v", err)}
	}

	// Build extensions from IDs
	extensions := buildExtensionsFromIDs(extIDs, curves, pointFormats)

	spec := &utls.ClientHelloSpec{
		TLSVersMin:         uint16(version),
		TLSVersMax:         uint16(version),
		CipherSuites:       ciphers,
		CompressionMethods: []uint8{0},
		Extensions:         extensions,
	}

	return BrowserProfile{
		Name:            "custom_ja3",
		ClientHelloID:   utls.HelloCustom,
		ClientHelloSpec: spec,
	}, nil
}

func parseDashSeparatedUint16(s string) ([]uint16, error) {
	if s == "" {
		return nil, nil
	}
	parts := strings.Split(s, "-")
	result := make([]uint16, 0, len(parts))
	for _, p := range parts {
		v, err := strconv.ParseUint(p, 10, 16)
		if err != nil {
			return nil, fmt.Errorf("parse %q: %w", p, err)
		}
		result = append(result, uint16(v))
	}
	return result, nil
}

func parseDashSeparatedUint8(s string) ([]uint8, error) {
	if s == "" {
		return nil, nil
	}
	parts := strings.Split(s, "-")
	result := make([]uint8, 0, len(parts))
	for _, p := range parts {
		v, err := strconv.ParseUint(p, 10, 8)
		if err != nil {
			return nil, fmt.Errorf("parse %q: %w", p, err)
		}
		result = append(result, uint8(v))
	}
	return result, nil
}

// isGREASE checks if a uint16 matches the GREASE pattern (0x?a?a).
func isGREASE(v uint16) bool {
	return (v & 0x0f0f) == 0x0a0a
}

func filterGREASE(vals []uint16) []uint16 {
	filtered := make([]uint16, 0, len(vals))
	for _, v := range vals {
		if !isGREASE(v) {
			filtered = append(filtered, v)
		}
	}
	return filtered
}

// buildExtensionsFromIDs creates uTLS extension objects from extension type IDs.
// This maps known extension IDs to their uTLS types; unknown IDs get a generic extension.
func buildExtensionsFromIDs(extIDs []uint16, curves []uint16, pointFormats []uint8) []utls.TLSExtension {
	exts := make([]utls.TLSExtension, 0, len(extIDs))

	curveIDs := make([]utls.CurveID, len(curves))
	for i, c := range curves {
		curveIDs[i] = utls.CurveID(c)
	}

	for _, id := range extIDs {
		switch id {
		case 0: // SNI
			exts = append(exts, &utls.SNIExtension{})
		case 5: // status_request
			exts = append(exts, &utls.StatusRequestExtension{})
		case 10: // supported_groups
			exts = append(exts, &utls.SupportedCurvesExtension{Curves: curveIDs})
		case 11: // ec_point_formats
			exts = append(exts, &utls.SupportedPointsExtension{SupportedPoints: pointFormats})
		case 13: // signature_algorithms
			exts = append(exts, &utls.SignatureAlgorithmsExtension{
				SupportedSignatureAlgorithms: []utls.SignatureScheme{
					utls.ECDSAWithP256AndSHA256,
					utls.PSSWithSHA256,
					utls.PKCS1WithSHA256,
					utls.ECDSAWithP384AndSHA384,
					utls.PSSWithSHA384,
					utls.PKCS1WithSHA384,
					utls.PSSWithSHA512,
					utls.PKCS1WithSHA512,
				},
			})
		case 16: // ALPN
			exts = append(exts, &utls.ALPNExtension{AlpnProtocols: []string{"h2", "http/1.1"}})
		case 18: // SCT
			exts = append(exts, &utls.SCTExtension{})
		case 23: // extended_master_secret
			exts = append(exts, &utls.ExtendedMasterSecretExtension{})
		case 27: // compress_certificate
			exts = append(exts, &utls.UtlsCompressCertExtension{
				Algorithms: []utls.CertCompressionAlgo{utls.CertCompressionBrotli},
			})
		case 35: // session_ticket
			exts = append(exts, &utls.SessionTicketExtension{})
		case 43: // supported_versions
			exts = append(exts, &utls.SupportedVersionsExtension{
				Versions: []uint16{utls.VersionTLS13, utls.VersionTLS12},
			})
		case 45: // psk_key_exchange_modes
			exts = append(exts, &utls.PSKKeyExchangeModesExtension{
				Modes: []uint8{utls.PskModeDHE},
			})
		case 51: // key_share
			shares := make([]utls.KeyShare, 0, len(curveIDs))

			exts = append(exts, &utls.RenegotiationInfoExtension{
				Renegotiation: utls.RenegotiateOnceAsClient,
			})
		default:
			exts = append(exts, &utls.GenericExtension{Id: id})
		}
	}

	return exts
}
```

```go
package tlsfetch

import (
	"fmt"
	"strings"
)

// ValidationReport contains the calculated fingerprints for a profile.
type ValidationReport struct {
	Name           string
	H2Fingerprint  string
	PseudoHeaders  string
	SettingsCount  int
}

// CalculateH2Fingerprint returns the Akamai-style HTTP/2 fingerprint string
// for a browser profile.
//
// Format: SETTINGS|WINDOW_UPDATE|PRIORITY_COUNT|PSEUDO_HEADER_ORDER
// Example: 1:65536;2:0;4:6291456;6:262144|15663105|0|m,a,s,p
func CalculateH2Fingerprint(p BrowserProfile) string {
	// SETTINGS
	settingsParts := make([]string, len(p.H2Settings))
	for i, s := range p.H2Settings {
		settingsParts[i] = fmt.Sprintf("%d:%d", s.ID, s.Value)
	}
	settings := strings.Join(settingsParts, ";")

	// WINDOW_UPDATE
	windowUpdate := fmt.Sprintf("%d", p.H2WindowUpdate)

	// PRIORITY count
	priorityCount := fmt.Sprintf("%d", len(p.H2Priorities))

	// Pseudo-header order as shorthand
	phParts := make([]string, 4)
	for i, ph := range p.PseudoHeaderOrder {
		phParts[i] = pseudoHeaderShorthand(ph)
	}
	pseudoHeaders := strings.Join(phParts, ",")

	return fmt.Sprintf("%s|%s|%s|%s", settings, windowUpdate, priorityCount, pseudoHeaders)
}

// pseudoHeaderShorthand converts a full pseudo-header name to its shorthand.
func pseudoHeaderShorthand(ph string) string {
	switch ph {
	case ":method":
		return "m"
	case ":authority":
		return "a"
	case ":scheme":
		return "s"
	case ":path":
		return "p"
	default:
		return ph
	}
}

// ValidateProfile calculates all fingerprints for a profile and returns
// a validation report. Use this to verify a profile produces the expected
// fingerprint before making requests.
func ValidateProfile(p BrowserProfile) ValidationReport {
	h2fp := CalculateH2Fingerprint(p)

	phParts := make([]string, 4)
	for i, ph := range p.PseudoHeaderOrder {
		phParts[i] = pseudoHeaderShorthand(ph)
	}

	return ValidationReport{
		Name:          p.Name,
		H2Fingerprint: h2fp,
		PseudoHeaders: strings.Join(phParts, ","),
		SettingsCount: len(p.H2Settings),
	}
}
```

```go
package h2

import (
	"encoding/binary"
	"io"
)

const (
	frameTypeSettings     = 0x04
	frameTypePriority     = 0x02
	frameTypeWindowUpdate = 0x08
)

const http2Preface = "PRI * HTTP/2.0\r\n\r\nSM\r\n\r\n"

// writeHTTP2Preface writes the HTTP/2 connection preface.
func writeHTTP2Preface(w io.Writer) {
	io.WriteString(w, http2Preface)
}

// writeFrameHeader writes a 9-byte HTTP/2 frame header.
func writeFrameHeader(w io.Writer, length int, frameType byte, flags byte, streamID uint32) error {
	var header [9]byte
	header[0] = byte(length >> 16)
	header[1] = byte(length >> 8)
	header[2] = byte(length)
	header[3] = frameType
	header[4] = flags
	binary.BigEndian.PutUint32(header[5:], streamID&0x7FFFFFFF)
	_, err := w.Write(header[:])
	return err
}

// writeSettingsFrame writes a SETTINGS frame with the given parameters in order.
func writeSettingsFrame(w io.Writer, settings []Setting) error {
	payloadLen := 6 * len(settings)
	if err := writeFrameHeader(w, payloadLen, frameTypeSettings, 0, 0); err != nil {
		return err
	}

	for _, s := range settings {
		var param [6]byte
		binary.BigEndian.PutUint16(param[0:2], s.ID)
		binary.BigEndian.PutUint32(param[2:6], s.Value)
		if _, err := w.Write(param[:]); err != nil {
			return err
		}
	}
	return nil
}

// writeWindowUpdateFrame writes a WINDOW_UPDATE frame.
func writeWindowUpdateFrame(w io.Writer, streamID, increment uint32) error {
	if err := writeFrameHeader(w, 4, frameTypeWindowUpdate, 0, streamID); err != nil {
		return err
	}
	var payload [4]byte
	binary.BigEndian.PutUint32(payload[:], increment&0x7FFFFFFF)
	_, err := w.Write(payload[:])
	return err
}

// writePriorityFrame writes a PRIORITY frame.
func writePriorityFrame(w io.Writer, p Priority) error {
	if err := writeFrameHeader(w, 5, frameTypePriority, 0, p.StreamID); err != nil {
		return err
	}
	var payload [5]byte
	dep := p.DependsOn
	if p.Exclusive {
		dep |= 0x80000000
	}
	binary.BigEndian.PutUint32(payload[0:4], dep)
	payload[4] = p.Weight
	_, err := w.Write(payload[:])
	return err
}

// WriteInitialFrames writes the HTTP/2 connection preface followed by
// custom SETTINGS, WINDOW_UPDATE, and PRIORITY frames. This is called
// on a raw TLS connection before handing it to the HTTP/2 transport.
func WriteInitialFrames(w io.Writer, cfg Config) error {
	writeHTTP2Preface(w)

	if err := writeSettingsFrame(w, cfg.Settings); err != nil {
		return err
	}

	if cfg.WindowUpdate > 0 {
		if err := writeWindowUpdateFrame(w, 0, cfg.WindowUpdate); err != nil {
			return err
		}
	}

	for _, p := range cfg.Priorities {
		if err := writePriorityFrame(w, p); err != nil {
			return err
		}
	}

	return nil
}
```

```go
package h2

import (
	"crypto/tls"
	"net"
	"net/http"
	"sync"

	"golang.org/x/net/http2"
)

// Transport is an HTTP/2 transport that applies custom fingerprint settings.
// It wraps the standard http2.Transport and manages connection setup.
type Transport struct {
	config Config

	mu    sync.Mutex
	conns map[string]*http2.ClientConn // addr → client conn
}

// NewTransport creates an HTTP/2 transport with custom fingerprint settings.
func NewTransport(cfg Config) *Transport {
	return &Transport{
		config: cfg,
		conns:  make(map[string]*http2.ClientConn),
	}
}

// RoundTrip sends an HTTP/2 request using a connection with custom framing.
// The tlsConn must already have completed the TLS handshake with ALPN "h2".
func (t *Transport) RoundTrip(req *http.Request, tlsConn *tls.Conn) (*http.Response, error) {
	addr := req.URL.Host

	t.mu.Lock()
	cc, ok := t.conns[addr]
	t.mu.Unlock()

	if ok {
		resp, err := cc.RoundTrip(req)
		if err == nil {
			return resp, nil
		}
		// Connection might be dead, remove and create new
		t.mu.Lock()
		delete(t.conns, addr)
		t.mu.Unlock()
	}

	// Create new HTTP/2 client connection
	cc, err := t.newClientConn(tlsConn)
	if err != nil {
		return nil, err
	}

	t.mu.Lock()
	t.conns[addr] = cc
	t.mu.Unlock()

	return cc.RoundTrip(req)
}

func (t *Transport) newClientConn(tlsConn net.Conn) (*http2.ClientConn, error) {
	// Write our custom HTTP/2 initial frames on the raw connection.
	// This happens BEFORE the standard http2.Transport touches the connection.
	if err := WriteInitialFrames(tlsConn, t.config); err != nil {
		return nil, err
	}

	// Create standard HTTP/2 transport configured to NOT write its own preface/settings
	// (since we already wrote ours).
	h2Transport := &http2.Transport{
		AllowHTTP:          false,
		DisableCompression: false,
	}

	// Use http2.Transport.NewClientConn to create a managed client connection
	// on the already-initialized TLS connection.
	cc, err := h2Transport.NewClientConn(tlsConn)
	if err != nil {
		return nil, err
	}

	return cc, nil
}

// CloseIdleConnections closes idle HTTP/2 connections.
func (t *Transport) CloseIdleConnections() {
	t.mu.Lock()
	defer t.mu.Unlock()

	for addr, cc := range t.conns {
		cc.Close()
		delete(t.conns, addr)
	}
}
```

**Important implementation note for the engineer:** The `http2.Transport.NewClientConn` method writes its own connection preface and SETTINGS. This means our custom frames and the standard ones will BOTH be sent. The approach above is a starting point. During implementation, you may need to:

1. Check if `http2.Transport` has a hook to skip its own preface (check the latest Go docs).
2. If not, wrap the `net.Conn` to absorb/discard the duplicate preface that `NewClientConn` writes, keeping only ours.
3. Alternatively, fork only the `newClientConn` function from `x/net/http2` into this package, which is ~100 lines.

The framing tests in Step 2 validate that our frames are correctly encoded regardless of which integration approach is used.

```go
// Transport is an http.RoundTripper that applies TLS and HTTP/2 fingerprinting.
// It returns standard *http.Response and accepts standard *http.Request.
type Transport struct {
	profile BrowserProfile
	dialer  proxy.Dialer
	pool    *internal.Pool
	h2t     *h2.Transport
	logger  *slog.Logger

	insecureSkipVerify bool

	mu          sync.Mutex

// NewTransport creates a fingerprinted HTTP transport.
func NewTransport(opts ...Option) (*Transport, error) {
	cfg := defaultConfig()
	for _, opt := range opts {
		opt(cfg)
	}

	profile, err := resolveProfile(cfg)
	if err != nil {
		return nil, err
	}

	var d proxy.Dialer
	if cfg.proxyURL != "" {
		u, err := url.Parse(cfg.proxyURL)
		if err != nil {
			return nil, fmt.Errorf("tlsfetch: parse proxy URL: %w", err)
		}
		d, err = proxy.FromURL(u)
		if err != nil {
			return nil, err
		}
	} else {
		d, _ = proxy.FromURL(nil)
	}

	h2Config := h2.Config{
		Settings:          make([]h2.Setting, len(profile.H2Settings)),
		WindowUpdate:      profile.H2WindowUpdate,
		PseudoHeaderOrder: profile.PseudoHeaderOrder,
	}
	for i, s := range profile.H2Settings {
		h2Config.Settings[i] = h2.Setting{ID: s.ID, Value: s.Value}
	}
	for _, p := range profile.H2Priorities {
		h2Config.Priorities = append(h2Config.Priorities, h2.Priority{
			StreamID:  p.StreamID,
			Exclusive: p.Exclusive,
			DependsOn: p.DependsOn,
			Weight:    p.Weight,
		})
	}

	pool := internal.NewPool(internal.PoolConfig{
		MaxConnsPerHost: cfg.pool.MaxConnsPerHost,
		MaxIdleConns:    cfg.pool.MaxIdleConns,
		IdleTimeout:     cfg.pool.IdleTimeout,
	})

	logger := cfg.logger
	if logger == nil {
		logger = slog.Default()
	}

	return &Transport{
		profile:            *profile,
		dialer:             d,
		pool:               pool,
		h2t:                h2.NewTransport(h2Config),
		logger:             logger,
		insecureSkipVerify: cfg.insecureSkipVerify,
	}, nil
}

// RoundTrip implements http.RoundTripper.
func (t *Transport) RoundTrip(req *http.Request) (*http.Response, error) {
	host := req.URL.Hostname()
	port := req.URL.Port()
	if port == "" {
		if req.URL.Scheme == "https" {
			port = "443"
		} else {
			port = "80"
		}
	}
	addr := net.JoinHostPort(host, port)

	// Dial and TLS handshake
	tlsConn, err := t.dialTLS(req.Context(), addr, host)
	if err != nil {
		return nil, err
	}

	// Check negotiated protocol
	state := tlsConn.ConnectionState()
	protocol := state.NegotiatedProtocol

	t.logger.Debug("tls handshake complete",
		"host", host,
		"protocol", protocol,
	)

	if protocol == "h2" {
		return t.h2t.RoundTrip(req, tlsConn.NetConn().(*net.TCPConn))
	}

	// HTTP/1.1 fallback using stdlib transport
	// Header ordering is applied by wrapping the connection
	return t.roundTripH1(req, tlsConn)
}

func (t *Transport) dialTLS(ctx context.Context, addr, serverName string) (*utls.UConn, error) {
	conn, err := t.dialer.DialContext(ctx, "tcp", addr)
	if err != nil {
		return nil, fmt.Errorf("tlsfetch: dial %s: %w", addr, err)
	}

	tlsConfig := &utls.Config{
		ServerName:         serverName,
		InsecureSkipVerify: t.insecureSkipVerify,
	}

	var uconn *utls.UConn
	if t.profile.ClientHelloSpec != nil {
		uconn = utls.UClient(conn, tlsConfig, utls.HelloCustom)
		if err := uconn.ApplyPreset(*t.profile.ClientHelloSpec); err != nil {
			conn.Close()
			return nil, fmt.Errorf("tlsfetch: apply TLS preset: %w", err)
		}
	} else {
		uconn = utls.UClient(conn, tlsConfig, t.profile.ClientHelloID)
	}

	if err := uconn.HandshakeContext(ctx); err != nil {
		conn.Close()
		return nil, fmt.Errorf("tlsfetch: TLS handshake with %s: %w", serverName, err)
	}

	return uconn, nil
}

func (t *Transport) roundTripH1(req *http.Request, tlsConn *utls.UConn) (*http.Response, error) {
	// For HTTP/1.1, we use a conn wrapper that reorders headers
	wrappedConn := header.NewReorderConn(tlsConn, t.profile.HeaderOrder)

	// Use a one-shot http.Transport with the pre-dialed connection
	tr := &http.Transport{
		DialTLSContext: func(ctx context.Context, network, addr string) (net.Conn, error) {
			return wrappedConn, nil
		},
		DisableKeepAlives: true,
	}

	return tr.RoundTrip(req)
}

// PoolStats returns connection pool statistics.
func (t *Transport) PoolStats() internal.PoolStats {
	return t.pool.Stats()
}

// CloseIdleConnections closes all idle connections.
func (t *Transport) CloseIdleConnections() {
	t.pool.Close()
	t.h2t.CloseIdleConnections()
}

// resolveProfile determines the BrowserProfile from config options.
func resolveProfile(cfg *config) (*BrowserProfile, error) {
	if cfg.profile != nil {
		return cfg.profile, nil
	}
	if cfg.profilePath != "" {
		p, err := LoadProfileFromFile(cfg.profilePath)
		if err != nil {
			return nil, err
		}
		return &p, nil
	}
	if cfg.ja3String != "" {
		p, err := ProfileFromJA3(cfg.ja3String)
		if err != nil {
			return nil, err
		}
		return &p, nil
	}
	return nil, fmt.Errorf("tlsfetch: no profile configured (use WithProfile, WithProfileFromFile, or WithProfileFromJA3)")
}
```

```go
// ReorderConn wraps a net.Conn and reorders HTTP/1.1 headers on Write.
type ReorderConn struct {
	net.Conn
	order       []string
	headersDone bool
	buf         bytes.Buffer
}

// NewReorderConn creates a conn wrapper that reorders HTTP/1.1 headers.
func NewReorderConn(conn net.Conn, order []string) *ReorderConn {
	return &ReorderConn{Conn: conn, order: order}
}

func (c *ReorderConn) Write(p []byte) (int, error) {
	if c.headersDone || len(c.order) == 0 {
		return c.Conn.Write(p)
	}

	c.buf.Write(p)

	// Check if we have the full header section
	if idx := bytes.Index(c.buf.Bytes(), headerEnd); idx != -1 {
		c.headersDone = true
		reordered := ReorderHTTP1Headers(c.buf.Bytes(), c.order)
		c.buf.Reset()
		return c.Conn.Write(reordered)
	}

	return len(p), nil
}
```

```go
		fmt.Fprintf(os.Stderr, "Error: %v\n", err)
		os.Exit(1)
	}
	defer resp.Body.Close()

	if *verbose {
		fmt.Fprintf(os.Stderr, "Status: %s\n", resp.Status)
		fmt.Fprintf(os.Stderr, "Proto: %s\n", resp.Proto)
		fmt.Fprintln(os.Stderr, "---")
	}

	io.Copy(os.Stdout, resp.Body)
}

func resolveProfileByName(name string) tlsfetch.BrowserProfile {
	switch name {
	case "chrome131":
		return tlsfetch.Chrome131
	case "firefox128":
		return tlsfetch.Firefox128
	case "safari18":
		return tlsfetch.Safari18
	case "edge131":
		return tlsfetch.Edge131
	default:
		fmt.Fprintf(os.Stderr, "Unknown profile: %s\n", name)
		os.Exit(1)
		return tlsfetch.BrowserProfile{}
	}
}

func resolveProfileOption(name, filePath string) tlsfetch.Option {
	if filePath != "" {
		return tlsfetch.WithProfileFromFile(filePath)
	}
	return tlsfetch.WithProfile(resolveProfileByName(name))
}
```
